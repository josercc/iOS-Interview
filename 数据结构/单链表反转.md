### 示例代码

```swift
func reverseNode(_ node:Node?) -> Node? {
    guard let head = node, let next = node?.next else {
        return node
    }
    var pre:Node? = head
    var cur:Node? = next
    var temp:Node? = next.next
    while cur != nil {
        temp = cur?.next
        cur?.next = pre
        pre = cur
        cur = temp
    }
    head.next = nil
    return pre
}
```

```swift
func reverseNode(_ node:Node?) -> Node? {
    guard let head = node, let next = node?.next else {
        return node
    }
    let newNode = reverseNode(next)
    head.next?.next = head
    head.next = nil
    return newNode
}
```
