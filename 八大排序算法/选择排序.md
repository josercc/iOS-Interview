## 示例代码

```swift
class Solution {
    /// 选择排序
    func selectSort(_ numbers:inout [Int]) {
        guard numbers.count > 1 else {
            return
        }
        for i in 0 ..< (numbers.count - 1) {
            var min = numbers[i]
            var minIndex = i
            var j = i + 1
            while j < numbers.count {
                let n1 = numbers[j]
                if n1 < min {
                    min = n1
                    minIndex = j
                }
                j += 1
            }
            numbers[minIndex] = numbers[i]
            numbers[i] = min
        }
    }
}
```
