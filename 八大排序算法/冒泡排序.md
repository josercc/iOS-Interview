## 示例代码

```swift
class Solution {
    func bubble(_ numbers:inout [Int]) {
        guard numbers.count > 1 else {
            return
        }
        var count = numbers.count
        while count > 1 {
            for i in 0 ..< (count - 1) {
                let j = i + 1
                let n1 = numbers[i]
                let n2 = numbers[j]
                if n2 < n1 {
                    numbers[i] = n2
                    numbers[j] = n1
                }
                print(numbers)
            }
            count -= 1
        }
    }
}
```
